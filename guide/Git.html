# 初次建仓

# 创建仓库（项目目录）
mkdir project
cd project
# 本地初始化
git init
git add .
git commit -m ""
# 连接远程仓库，创建主分支
git remote add origin+远程仓库地址
# 把本地仓库的变化连接到远程仓库主分支 -u指定origin为默认主机
git push -u origin master

# 更新本地到远程

# 添加缓存
git add 文件
# 提交信息
git commit -m "msg"
# 推送（先更新） origin主机名
git push origin 本地分支:远程分支
# 本地和远程分支同名
git push origin
# 只有一个主机
git push

# 拉取远程到本地

# 拉取到本地
git fetch
# 拉取到本地，并合并本地分支
git pull
git pull <远程主机> <远程分支>:<本地分支>
# 省略本地分支
git pull origin master
# 如果远程冲突，将冲突文件放到暂存区(新分支)
git pull -r
