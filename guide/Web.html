# ajax 提交表单

<form action="" method="POST" name="base-form" id="base-form">
    <button type="button" class="btn btn-dark handle"> 提交</button>
</form>

<!-- msg Modal -->
<div class="modal fade" id="success-modal" tabindex="-1" role="dialog" aria-labelledby="modal-block-fadein" aria-hidden="true">
    <div class="modal-dialog" role="document">
        <div class="modal-content">
            <div class="alert alert-success alert-dismissable" role="alert">
                <p class="mb-0">操作成功</p>
            </div>
        </div>
    </div>
</div>
<div class="modal fade" id="error-modal" tabindex="-1" role="dialog" aria-labelledby="modal-block-fadein" aria-hidden="true">
    <div class="modal-dialog" role="document">
        <div class="modal-content">
            <div class="alert alert-danger alert-dismissable" role="alert">
                <p class="mb-0" id="msg">操作失败</p>
            </div>
        </div>
    </div>
</div>
<!-- msg Modal -->

<script>
    $(function () {
        $(".handle").click(function () {
            $.ajax({
                url: "{:admin_url('user/update')}",
                data: $("#base-form").serialize(),
                dataType: "JSON",
                method: "POST",
                success: function (data) {
                    if (data.code == 1) {
                        $("#success-modal").modal("show");
                        window.history.go(-1);
                    } else {
                        $("#msg").text(data.msg);
                        $("#error-modal").modal("show");
                        setTimeout("$('#error-modal').modal('hide')", 2000)
                    }
                }
            });
        });
    });
</script>
